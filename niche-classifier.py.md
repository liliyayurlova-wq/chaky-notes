def classify_niche(client_input: str) -> dict:
    """
    Определяет нишу по ключевым словам из описания клиента.
    Поиск мягкий: ['женские', 'футболк'] ищет и 'женские футболки' и 'футболки женские'.
    Возвращает: {niche: str, category: str, keywords: list, confidence: float}
    """
    input_lower = client_input.lower()
    found = []

    for niche, keywords in MPSTATS_CATEGORIES.items():
        for kw in keywords:
            # Разбиваем ключ на слова и проверяем каждое
            kw_parts = kw.lower().split()
            # Мягкий поиск: если хотя бы 60% слов ключа есть в тексте
            if len(kw_parts) == 0:
                continue
            matched = sum(1 for part in kw_parts if part in input_lower)
            if len(kw_parts) > 0 and matched / len(kw_parts) >= 0.6:
                found.append((niche, kw))

    # Если ничего не нашли — пробуем одиночные слова
    if not found:
        single_keywords = {
            "футболк": ["мужская одежда", "женская одежда"],
            "плать": ["женская одежда"],
            "джинс": ["женская одежда", "мужская одежда"],
            "кроссовк": ["женская обувь", "мужская обувь"],
            "игрушк": ["детские игрушки"],
            "косметик": ["косметика"],
            "крем": ["косметика"],
            "шампун": ["косметика"],
            "одежд": ["женская одежда", "мужская одежда", "детская одежда"],
            "обув": ["женская обувь", "мужская обувь"],
            "сумк": ["сумки"],
            "бель": ["белье"],
            "наушник": ["наушники"],
            "техник": ["электроника", "кухонная техника"],
            "инструмент": ["инструменты", "электрика"],
            "мебель": ["мебель", "офисная мебель"],
            "корм": ["товары для кошек", "товары для собак"],
            "спортив": ["спорт", "спортивная одежда", "спортивное питание"],
            "строительн": ["инструменты", "электрика"],
            "авто": ["автотовары"],
            "бад": ["бады"],
            "витамин": ["бады"],
            "светильн": ["освещение"],
            "туризм": ["туризм и отдых"],
            "палатк": ["туризм и отдых"],
            "детск": ["детские игрушки", "детская одежда", "товары для детей"],
            "трусы": ["белье"],
            "носк": ["белье"],
        }
        for word, niches in single_keywords.items():
            if word in input_lower:
                for n in niches:
                    found.append((n, word))
                break  # первое совпадение

    if found:
        niches_found = list(dict.fromkeys([f[0] for f in found]))  # order-preserving unique
        keywords_found = list(dict.fromkeys([f[1] for f in found]))
        confidence = min(1.0, len(niches_found) / 3)
        return {
            "niches": niches_found[:3],
            "keywords": keywords_found[:5],
            "confidence": round(confidence, 2)
        }

    return {
        "niches": ["не определено"],
        "keywords": [],
        "confidence": 0.0
    }


def extract_competitors(client_input: str) -> list:
    """
    Ищет упоминания конкурентов в тексте клиента.
    Паттерны: "конкуренты: X, Y", "конкурирую с X", "бренд X", "продавец X"
    """
    patterns = [
        r"(?:конкурент[а-я]*|конкуриру[юя][\w]*(?:с)?)\s*[:\s]+([А-ЯЁA-Z][А-ЯЁA-Zа-яёa-z0-9\s,\.]+)",
        r"(?:конкурент[а-я]*|конкуриру[юя][\w]*[:\s]+(?:с)?\s+)([А-ЯЁA-Z][а-яёa-zА-ЯЁA-Z0-9\s]+)",
        r"(?:мой\s+)?(?:основн[ыой]+\s+)?конкурент[а-я]*\s*[:\s]+([А-ЯЁA-Z][а-яёa-zА-ЯЁA-Z0-9\s,\.]+)",
    ]

    all_text = client_input
    competitors = []

    for p in patterns:
        matches = re.findall(p, all_text)
        for m in matches:
            parts = re.split(r'[,;]|\s+и\s+|\s+также\s+', m)
            for part in parts:
                part = part.strip().strip('. ,')
                if part and len(part) > 2 and not re.search(r'^(конкурент|бренд|продавец|не\s)', part, re.IGNORECASE):
                    if part not in competitors:
                        competitors.append(part)

    # Также проверяем по ключевым словам "конкуренты" в конце предложения
    if not competitors:
        # Строка вида "Конкуренты: Узкоттон, Мэлон"
        m = re.search(r'(?:конкурент[а-я]*|основной\s+конкурент)[\s:]+([А-ЯЁA-Z][а-яёa-zА-ЯЁA-Z\s,\.]+(?:[,\.]|$))', all_text, re.IGNORECASE)
        if m:
            parts = re.split(r'[,;]|\s+и\s+', m.group(1))
            for part in parts:
                part = part.strip().strip('. ,')
                if part and len(part) > 2 and part not in competitors:
                    competitors.append(part)

    # Фильтруем мусор
    stop_words = ["не знаю", "нет", "не указаны", "не помню", "не конкурент", "не"]
    competitors = [c for c in competitors if c.lower().strip() not in stop_words]

    return competitors


def determine_product_type(niches: list, keywords: list) -> str:
    """
    Определяет тип товара (импульсный/рациональный) по нише.
    """
    impulse_niches = [
        "бижутерия", "сумки", "игрушки", "аксессуары", "бельё",
        "носки", "колготки", "косметика", "парфюмерия", "декор",
        "свечи", "чехлы", "наушники"
    ]
    rational_niches = [
        "электроника", "мебель", "техника", "строительство",
        "автотовары", "здоровье", "бады", "спорт", "туризм",
        "электрика", "освещение", "инструменты"
    ]

    for niche in niches:
        if niche in rational_niches:
            return "рациональный"
        if niche in impulse_niches:
            return "импульсный"

    # По цене если не определили — по ключевым словам
    for kw in keywords:
        if any(w in kw for w in ["дорог", "премиум", "элит", "профессиональн"]):
            return "рациональный"
    return "импульсный"  # default


def get_price_segment(product_type: str, client_input: str) -> str:
    """Примерная оценка ценового сегмента по тексту"""
    if "дорог" in client_input.lower() or "премиум" in client_input.lower():
        return "высокий"
    if "дешев" in client_input.lower() or "бюджет" in client_input.lower():
        return "низкий"
    return "средний"
